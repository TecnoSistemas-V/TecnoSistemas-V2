Opción 3: Script para eliminar duplicados (Avanzado)
Si quieres una solución más técnica, este script elimina automáticamente todos los duplicados por cédula:

javascript
// Pega esto en la consola del navegador (F12) desde admin-lista.html
import { db } from './js/conexion.js';
import { collection, getDocs, deleteDoc, doc } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

const snap = await getDocs(collection(db, "estudiantes"));
const vistos = new Map();

for (const docSnap of snap.docs) {
    const cedula = docSnap.data().Cédula;
    if (vistos.has(cedula)) {
        await deleteDoc(doc(db, "estudiantes", docSnap.id));
        console.log(`Eliminado duplicado: ${cedula}`);
    } else {
        vistos.set(cedula, true);
    }
}
console.log("✅ Duplicados eliminados");
